## Widget Layout Code (Pseudo)

```python
# In UIManager.create_widgets()

def create_orders_file_section(self):
    """Creates the Orders file selection section with folder support."""

    group_box = QGroupBox("ğŸ“¦ Orders File")
    layout = QVBoxLayout()

    # Mode selector
    mode_layout = QHBoxLayout()
    mode_label = QLabel("Load Mode:")
    self.orders_mode_selector = QComboBox()
    self.orders_mode_selector.addItems(["Single File", "Folder (Multiple Files)"])
    self.orders_mode_selector.currentIndexChanged.connect(self.on_orders_mode_changed)

    mode_layout.addWidget(mode_label)
    mode_layout.addWidget(self.orders_mode_selector)
    mode_layout.addStretch()

    layout.addLayout(mode_layout)

    # File/Folder selection button
    self.orders_select_button = QPushButton("Select File...")
    self.orders_select_button.clicked.connect(self.on_orders_select_clicked)
    layout.addWidget(self.orders_select_button)

    # Selected path label
    path_layout = QHBoxLayout()
    path_layout.addWidget(QLabel("Selected:"))
    self.orders_file_path_label = QLabel("No file loaded")
    self.orders_file_status_label = QLabel("")  # âœ“ or âœ—
    path_layout.addWidget(self.orders_file_path_label)
    path_layout.addWidget(self.orders_file_status_label)
    path_layout.addStretch()

    layout.addLayout(path_layout)

    # File list widget (only shown in folder mode)
    self.orders_file_list_widget = QListWidget()
    self.orders_file_list_widget.setMaximumHeight(150)
    self.orders_file_list_widget.setVisible(False)  # Hidden by default
    layout.addWidget(self.orders_file_list_widget)

    # File count label (only shown in folder mode)
    self.orders_file_count_label = QLabel("")
    self.orders_file_count_label.setVisible(False)
    layout.addWidget(self.orders_file_count_label)

    # Options (only shown in folder mode)
    self.orders_options_widget = QWidget()
    options_layout = QVBoxLayout()
    self.orders_recursive_checkbox = QCheckBox("Include subfolders")
    self.orders_remove_duplicates_checkbox = QCheckBox("Remove duplicate orders")
    self.orders_remove_duplicates_checkbox.setChecked(True)  # Default: remove duplicates
    options_layout.addWidget(self.orders_recursive_checkbox)
    options_layout.addWidget(self.orders_remove_duplicates_checkbox)
    self.orders_options_widget.setLayout(options_layout)
    self.orders_options_widget.setVisible(False)  # Hidden by default

    layout.addWidget(self.orders_options_widget)

    group_box.setLayout(layout)
    return group_box


def on_orders_mode_changed(self, index):
    """Handle mode change between Single File and Folder."""
    is_folder_mode = (index == 1)

    # Update button text
    if is_folder_mode:
        self.orders_select_button.setText("Select Folder...")
    else:
        self.orders_select_button.setText("Select File...")

    # Show/hide folder-specific widgets
    self.orders_file_list_widget.setVisible(is_folder_mode)
    self.orders_file_count_label.setVisible(is_folder_mode)
    self.orders_options_widget.setVisible(is_folder_mode)


def on_orders_select_clicked(self):
    """Handle click on Select button (adapts based on mode)."""
    mode = self.orders_mode_selector.currentText()

    if mode == "Single File":
        self.file_handler.select_orders_file()  # Existing
    else:
        self.file_handler.select_orders_folder()  # NEW
```
